Tips for Docker

# See what docker images/containers are running:

sudo docker ps

# Get into the Linux Terminal of the Docker Container

sudo docker exec -it /bin/bash

OR

sudo docker exec -it /bin/sh

# Now that you are in the docker container/image, see what Linux OS is running:

cat /etc/os-release

# Exit the docker container terminal:

exit
